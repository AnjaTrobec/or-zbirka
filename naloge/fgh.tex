\begin{vprasanje}
Zapiši naslednji problem kot celoštevilski linearni program.
\begin{align*}
\min & f(x) + g(y) + h(z) \quad \text{p.p.} \\
f(x) &= \begin{cases}
        10      & \text{če $x = 0$, in} \\
        10 + 5x & \text{če $x > 0$}
        \end{cases} \\
g(y) &= \begin{cases}
        10     & \text{če $y = 0$, in} \\
        5 + 5y & \text{če $y > 0$}
        \end{cases} \\
h(z) &= \begin{cases}
        8       & \text{če $z = 0$, in} \\
        10 + 4z & \text{če $z > 0$}
        \end{cases} \\
x + y + z &\ge 100 \\
      (x \ge 5 \land y \ge 5)
&\lor (x \ge 5 \land z \ge 5)
 \lor (y \ge 5 \land z \ge 5) \\
x, y, z &\ge 0, \quad x, y, z \in \Z
\end{align*}
\end{vprasanje}
