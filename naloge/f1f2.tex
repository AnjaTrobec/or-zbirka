\begin{vprasanje}
Zapiši naslednji problem kot celoštevilski linearni program.
\begin{align*}
\min & f_1(x_1) + f_2(x_2) \quad \text{p.p.} \\
f_1(x_1) &= \begin{cases}
            0         & \text{če $x_1 = 0$, in} \\
            10 + 5x_1 & \text{če $x_1 > 0$}
            \end{cases} \\
f_2(x_2) &= \begin{cases}
            0         & \text{če $x_2 = 0$, in} \\
            20 + 4x_2 & \text{če $x_2 > 0$}
            \end{cases} \\
100x_1 + 150x_2 &\le 10000 \\
x_1, x_2 &\ge 0
\end{align*}
\end{vprasanje}
